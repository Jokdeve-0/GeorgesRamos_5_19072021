        var add_style = page == "valid"?`class="style_valid"`:""

            <div class="line">
                ${add_icon_remove}
                <p><img src="${product.imageUrl}" alt=""></p>
                <p>${product.name}</p>
                <p ${add_style}>${product.price}€</p>
            </div>`
        return boxBasket
    }
}
